    <!-- Application Structure Plan: The SPA is structured around an interactive 'Explorer Profile' system. Users select an explorer (Columbus, etc.) from a top navigation bar. This displays a dedicated profile card for that explorer, avoiding the cognitive load of a large table. Each profile contains input fields for the required research points (Motives, Achievements, etc.). A separate 'Comparison' view synthesizes user input, notably visualizing the 'Legacy Judgment' choices in a dynamic bar chart. This focused, one-at-a-time approach mirrors a 'gallery walk' station and promotes deeper engagement with each figure before moving to synthesis, which is more effective for student learning than a static grid. A new "Export to PDF" button allows comprehensive output of all filled data and the comparison chart. -->

    <!-- Visualization & Content Choices: Report Info: Explorer research prompts -> Goal: Organize & Compare -> Viz/Presentation: Interactive profile cards with textarea inputs and radio buttons for judgment. A dynamic Chart.js bar chart for comparison. A new "Export to PDF" button triggers client-side PDF generation of all explorer data and the comparison chart. -> Interaction: Users select an explorer to view/edit their profile. Input is saved automatically. The 'Legacy Judgment' radio button choices dynamically update the comparison chart. PDF export enables easy submission and grading of collected data. -> Justification: Profile cards focus the student's task. Radio buttons are better UX for single-choice questions than checkboxes. The dynamic chart provides immediate visual feedback and a synthesis of the student's own analysis, making the learning process interactive and reinforcing. PDF export enables easy submission and grading of collected data. -> Library/Method: Vanilla JS for state and DOM manipulation, Chart.js (Canvas) for the dynamic summary visualization, jsPDF for PDF document creation. -->

                da_gama: { name: 'da Gama', motives: '', achievement: '', consequence: '', quote: '', legacy: null, explanation: '' },

                magellan: { name: 'Magellan', motives: '', achievement: '', consequence: '', quote: '', legacy: null, explanation: '' },

                cortes: { name: 'Cortés', motives: '', achievement: '', consequence: '', quote: '', legacy: null, explanation: '' },

                de_soto: { name: 'de Soto', motives: '', achievement: '', consequence: '', quote: '', legacy: null, explanation: '' },

                vespucci: { name: 'Vespucci', motives: '', achievement: '', consequence: '', quote: '', legacy: null, explanation: '' }

            };


            const sections = [

                { id: 'motives', title: '1. Motives', prompt: 'What were they hoping to gain or prove by exploring? Look for actions and quotes that reveal purpose.' },

                { id: 'achievement', title: '2. Major Achievement', prompt: 'What did this explorer accomplish that made them famous or historically important?' },

                { id: 'consequence', title: '3. Harmful Consequence', prompt: 'Identify at least one serious negative effect of their actions. Who did it affect?' },

                { id: 'quote', title: '4. Primary Source Quote', prompt: 'Copy one quote from the text that reveals something important about their mindset or values.' },

                { id: 'legacy', title: '5. Legacy Judgment', prompt: 'How should this explorer be remembered? Explain your choice in 2–3 sentences.' }

            ];

            

            const legacyOptions = ['Hero', 'Complex Figure', 'Conqueror'];


            const mainContent = document.getElementById('main-content');

            const navButtons = document.querySelectorAll('.nav-button');

            function renderExplorerView(explorerId) {

                const explorer = explorerData[explorerId];

                let html = `<div id="${explorerId}-card" class="content-card active bg-white p-6 sm:p-8 rounded-xl shadow-lg animate-fade-in">`;

                html += `<h2 class="text-3xl font-bold mb-6 text-slate-700">${explorer.name}</h2>`;

                

                sections.forEach(section => {

                    html += `<div class="mb-6">

                                <h3 class="text-xl font-bold text-gray-800">${section.title}</h3>

                                <p class="text-gray-600 italic mb-3">${section.prompt}</p>`;

                    

                    if (section.id === 'legacy') {

                        html += '<div class="flex flex-col sm:flex-row gap-4 mb-3">';

                        legacyOptions.forEach(option => {

                            const isChecked = explorer.legacy === option;

                            html += `<label class="flex items-center space-x-2 cursor-pointer">

                                        <input type="radio" name="${explorerId}-legacy" value="${option}" class="form-radio h-5 w-5 text-slate-600" ${isChecked ? 'checked' : ''} data-explorer="${explorerId}" data-field="legacy">

                                        <span>${option}</span>

                                     </label>`;

                        });

                        html += '</div>';

                        html += `<textarea data-explorer="${explorerId}" data-field="explanation" class="w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-slate-400 focus:border-transparent transition" rows="3" placeholder="Explain your choice...">${explorer.explanation}</textarea>`;

                    } else {

                        html += `<textarea data-explorer="${explorerId}" data-field="${section.id}" class="w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-slate-400 focus:border-transparent transition" rows="4" placeholder="Your answer here...">${explorer[section.id]}</textarea>`;

                    }

                    html += `</div>`;

                });


                html += `</div>`;

                mainContent.innerHTML = html;

                attachInputListeners();

            }

            

            function renderComparisonView() {

                let html = `<div id="comparison-card" class="content-card active bg-white p-6 sm:p-8 rounded-xl shadow-lg animate-fade-in">

                                <h2 class="text-3xl font-bold mb-2 text-slate-700">Comparative Analysis</h2>

                                <p class="text-gray-600 mb-6">This chart summarizes your "Legacy Judgment" for each explorer. It updates automatically as you make your selections in each explorer's profile. This helps you visualize the overall impact and perception based on your research.</p>

                                <div class="chart-container">

                                    <canvas id="comparisonChart"></canvas>

                                </div>

                            </div>`;

                mainContent.innerHTML = html;


                const ctx = document.getElementById('comparisonChart').getContext('2d');

                const chartData = {

                    labels: legacyOptions,

                    datasets: [{

                        label: 'Number of Explorers',

                        data: [0, 0, 0],

                        backgroundColor: [

                            'rgba(75, 192, 192, 0.6)', /* Hero - Teal */

                            'rgba(255, 205, 86, 0.6)', /* Complex Figure - Yellow */

                            'rgba(255, 99, 132, 0.6)'  /* Conqueror - Red */

                        ],

                        borderColor: [

                            'rgba(75, 192, 192, 1)',

                            'rgba(255, 205, 86, 1)',

                            'rgba(255, 99, 132, 1)'

                        ],

                        borderWidth: 1

                    }]

                };


                if (comparisonChart) {

                    comparisonChart.destroy();

                }


                comparisonChart = new Chart(ctx, {

                    type: 'bar',

                    data: chartData,

                    options: {

                        responsive: true,

                        maintainAspectRatio: false,

                        scales: {

                            y: {

                                beginAtZero: true,

                                ticks: {

                                    stepSize: 1

                                }

                            }

                        },

                        plugins: {

                            legend: {

                                display: false

                            },

                            title: {

                                display: true,

                                text: 'Legacy Judgments',

                                font: {

                                    size: 18,

                                    family: 'Lora'

                                }

                            }

                        }

                    }

                });

                updateChartData();

            }

            

            function updateChartData() {

                if (!comparisonChart) return;


                const counts = { 'Hero': 0, 'Complex Figure': 0, 'Conqueror': 0 };

                Object.values(explorerData).forEach(explorer => {

                    if (explorer.legacy && counts.hasOwnProperty(explorer.legacy)) {

                        counts[explorer.legacy]++;

                    }

                });

                

                comparisonChart.data.datasets[0].data = legacyOptions.map(option => counts[option]);

                comparisonChart.update();

            }


            async function generatePdf() {

                const { jsPDF } = window.jspdf;

                const doc = new jsPDF('p', 'mm', 'a4'); // 'p' for portrait, 'mm' for millimeters, 'a4' size

                const margin = 15;

                let yOffset = margin;

                const lineHeight = 5;

                const headerFontSize = 16;

                const subHeaderFontSize = 12;

                const normalFontSize = 10;

                const pageHeight = doc.internal.pageSize.height;


                doc.setFont('helvetica', 'bold');

                doc.setFontSize(headerFontSize);

                doc.text('Explorer Gallery Walk Report', margin, yOffset);

                yOffset += lineHeight * 2;


                // Add data for each explorer

                for (const explorerId in explorerData) {

                    const explorer = explorerData[explorerId];

                    if (yOffset > pageHeight - margin - (lineHeight * 5)) { // Check if new page is needed

                        doc.addPage();

                        yOffset = margin;

                    }

                    doc.setFont('helvetica', 'bold');

                    doc.setFontSize(subHeaderFontSize);

                    doc.text(`${explorer.name} - Explorer Profile`, margin, yOffset);

                    yOffset += lineHeight * 2;


                    doc.setFont('helvetica', 'normal');

                    doc.setFontSize(normalFontSize);


                    sections.forEach(section => {

                        let content = explorer[section.id];

                        if (section.id === 'legacy') {

                            content = `Choice: ${explorer.legacy || 'N/A'}\nExplanation: ${explorer.explanation || 'N/A'}`;

                        } else if (!content) {

                            content = 'No answer provided.';

                        }


                        const textLines = doc.splitTextToSize(`${section.title}: ${content}`, doc.internal.pageSize.width - 2 * margin);

                        

                        if (yOffset + (textLines.length * lineHeight) > pageHeight - margin) {

                            doc.addPage();

                            yOffset = margin;

                        }

                        

                        doc.setFont('helvetica', 'bold');

                        doc.text(`${section.title}:`, margin, yOffset);

                        yOffset += lineHeight;

                        doc.setFont('helvetica', 'normal');

                        doc.text(textLines.slice(1), margin + 5, yOffset); // indent multi-line content

                        yOffset += (textLines.length - 1) * lineHeight;

                        if (textLines.length === 1) yOffset += lineHeight; // Add line for single line content


                        yOffset += lineHeight; // Spacer

                    });

                    yOffset += lineHeight * 2; // Spacer between explorers

                }


                // Add comparison chart

                if (comparisonChart) {

                    doc.addPage();

                    yOffset = margin;

                    doc.setFont('helvetica', 'bold');

                    doc.setFontSize(subHeaderFontSize);

                    doc.text('Legacy Judgments Comparison', margin, yOffset);

                    yOffset += lineHeight * 2;


                    const imgData = comparisonChart.toBase64Image('image/png', 1);

                    const imgWidth = 180; // mm

                    const imgHeight = (comparisonChart.canvas.height * imgWidth) / comparisonChart.canvas.width;


                    if (yOffset + imgHeight > pageHeight - margin) {

                        doc.addPage();

                        yOffset = margin;

                    }

                    doc.addImage(imgData, 'PNG', margin, yOffset, imgWidth, imgHeight);

                } else {

                    doc.addPage();

                    yOffset = margin;

                    doc.setFontSize(normalFontSize);

                    doc.text('Comparison chart not available for export.', margin, yOffset);

                }


                doc.save('Explorer_Gallery_Walk_Summary.pdf');

            }


            function handleNavClick(e) {

                const button = e.target.closest('.nav-button');

                if (!button) return;


                const explorerId = button.dataset.explorer;

                currentExplorer = explorerId;


                navButtons.forEach(btn => {

                    btn.classList.toggle('active', btn.dataset.explorer === explorerId);

                });


                if (explorerId === 'comparison') {

                    renderComparisonView();

                } else {

                    renderExplorerView(explorerId);

                }

            }


            function handleInputChange(e) {

                const element = e.target;

                const explorerId = element.dataset.explorer;

                const field = element.dataset.field;


                if (explorerData[explorerId] && field) {

                    explorerData[explorerId][field] = element.value;

                    if(field === 'legacy') {

                        updateChartData();

                    }

                }

            }

            

            function attachInputListeners() {

                const inputs = mainContent.querySelectorAll('textarea, input[type="radio"]');

                inputs.forEach(input => {

                    input.addEventListener('input', handleInputChange);

                    input.addEventListener('change', handleInputChange); // for radio buttons

                });

            }


            document.getElementById('explorer-nav').addEventListener('click', handleNavClick);

            exportPdfButton.addEventListener('click', generatePdf);


            // Initial render

            document.querySelector('.nav-button[data-explorer="columbus"]').classList.add('active');

            renderExplorerView('columbus');

        });
